// Unit Conversion Math Example
const KCAL_TO_KJ = 4.184;
const OZ_TO_GRAMS = 28.3495;

function processFood(entry) {
  // Convert Energy to kcal/100g
  let kcalPer100g = entry.rawEnergy;
  if (entry.energyUnit === 'kJ') {
    kcalPer100g = entry.rawEnergy / KCAL_TO_KJ;
  }

  // Convert Mass to grams
  let massGrams = entry.rawMass;
  if (entry.massUnit === 'oz') {
    massGrams = entry.rawMass * OZ_TO_GRAMS;
  }

  // Calculate Consumed Values
  let calories = (kcalPer100g * massGrams) / 100;
  let protein = (entry.proteinPer100g * massGrams) / 100;
  let carbs = (entry.carbsPer100g * massGrams) / 100;
  let fats = (entry.fatsPer100g * massGrams) / 100;

  return {
    calories: Math.round(calories),
    protein: Math.round(protein * 10) / 10,
    carbs: Math.round(carbs * 10) / 10,
    fats: Math.round(fats * 10) / 10,
    massGrams: Math.round(massGrams * 10) / 10
  };
}
